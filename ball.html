import pygame import sys # 初始化 pygame.init() # 顏色與視窗設定 WHITE = (255, 255, 255) BLACK = (0, 0, 0) WIDTH, HEIGHT = 800, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("單人桌球 vs AI (固定速度版)") # 常數設定 PADDLE_WIDTH, PADDLE_HEIGHT = 10, 100 BALL_SIZE = 15 BALL_SPEED = 6 # 固定球速 PLAYER_SPEED = 7 # 玩家移動速度 AI_SPEED = 5 # AI 移動速度 (稍微調慢一點，玩家才有機會贏) # 遊戲物件 player_paddle = pygame.Rect(20, HEIGHT//2 - PADDLE_HEIGHT//2, PADDLE_WIDTH, PADDLE_HEIGHT) ai_paddle = pygame.Rect(WIDTH - 30, HEIGHT//2 - PADDLE_HEIGHT//2, PADDLE_WIDTH, PADDLE_HEIGHT) ball = pygame.Rect(WIDTH//2, HEIGHT//2, BALL_SIZE, BALL_SIZE) ball_dx = BALL_SPEED ball_dy = BALL_SPEED # 分數 player_score = 0 ai_score = 0 font = pygame.font.SysFont("Arial", 32) clock = pygame.time.Clock() while True: for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() # 1. 玩家控制 (左側) keys = pygame.key.get_pressed() if (keys[pygame.K_w] or keys[pygame.K_UP]) and player_paddle.top > 0: player_paddle.y -= PLAYER_SPEED if (keys[pygame.K_s] or keys[pygame.K_DOWN]) and player_paddle.bottom < HEIGHT: player_paddle.y += PLAYER_SPEED # 2. AI 邏輯 (右側) # AI 會判斷球的高度與自己中心點的距離 if ai_paddle.centerY < ball.centerY: ai_paddle.y += AI_SPEED elif ai_paddle.centerY > ball.centerY: ai_paddle.y -= AI_SPEED # 確保 AI 不會超出螢幕 if ai_paddle.top < 0: ai_paddle.top = 0 if ai_paddle.bottom > HEIGHT: ai_paddle.bottom = HEIGHT # 3. 球的移動 ball.x += ball_dx ball.y += ball_dy # 4. 碰撞邏輯 # 頂部/底部反彈 if ball.top <= 0 or ball.bottom >= HEIGHT: ball_dy *= -1 # 擋板碰撞 if ball.colliderect(player_paddle) or ball.colliderect(ai_paddle): ball_dx *= -1 # 這裡不增加速度，確保符合你的要求 # 5. 計分與重置 if ball.left <= 0: ai_score += 1 ball.center = (WIDTH//2, HEIGHT//2) ball_dx = BALL_SPEED # 重置方向 if ball.right >= WIDTH: player_score += 1 ball.center = (WIDTH//2, HEIGHT//2) ball_dx = -BALL_SPEED # 6. 繪製畫面 screen.fill(BLACK) pygame.draw.rect(screen, WHITE, player_paddle) pygame.draw.rect(screen, WHITE, ai_paddle) pygame.draw.ellipse(screen, WHITE, ball) pygame.draw.aaline(screen, WHITE, (WIDTH//2, 0), (WIDTH//2, HEIGHT)) # 顯示分數 score_display = font.render(f"You: {player_score} AI: {ai_score}", True, WHITE) screen.blit(score_display, (WIDTH//2 - 100, 20)) pygame.display.flip() clock.tick(60)
